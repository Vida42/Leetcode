```python
class Solution(object):
	def closestValue(self, root, target):
		"""
		:type root: TreeNode
		:type target: float
		:rtype: int
		"""
		res = []
		self.traversal(root, res)
		return min([abs(i-target) for i in res])

	def traversal(self, node, res):
		if not node:
			return None
		res.append(node.val)
		self.traversal(node.left, res)
		self.traversal(node.right, res)
		return res
```
