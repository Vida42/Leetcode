```python
class Solution(object):
    def permute(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        if not nums:
            return res
        if len(nums) == 0:
            res.append([])
            return res
        self.helper(nums, [], res)
        return res

    def helper(self, nums, path, res):
        # 回溯的一步
        if len(path) == len(nums):
            res.append(path)
            return
        for i in range(len(nums)):
            # 判断
            if nums[i] not in path:
                # 下面是递归不变
                self.helper(nums, path+[nums[i]], res)
```
