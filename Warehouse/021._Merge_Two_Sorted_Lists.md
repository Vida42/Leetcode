```python
# 92 ms, faster than 13.86%

# Definition for singly-linked list.
class ListNode:
	 def __init__(self, x):
		 self.val = x
		 self.next = None

class Solution:
	def mergeTwoLists(self, l1, l2):
		"""
		:type l1: ListNode
		:type l2: ListNode
		:rtype: ListNode
		"""
		
		head = ListNode(0)
		move = head
		if not l1:
			return l2
		if not l2:
			return l1
		while l1 and l2:
			if l1.val < l2.val:
				move.next = l1
				l1 = l1.next
			else:
				move.next = l2
				l2 = l2.next
			move = move.next
		move.next = l1 if l1 else l2
		return head.next
```
