class Solution:
    def search(self, reader, target):
        """
        :type reader: ArrayReader
        :type target: int
        :rtype: int
        """
        if reader.get(0) == target:
            return 0
        lo, hi = 0, self.getRange(reader, target)
        while lo + 1 < hi:
            mid = (lo + hi)//2
            tmp = reader.get(mid)
            if tmp == target:
                return mid
            elif tmp < target:
                lo = mid
            else:
                hi = mid
        if reader.get(lo) == target:
            return lo
        if reader.get(hi) == target:
            return hi
        return -1
